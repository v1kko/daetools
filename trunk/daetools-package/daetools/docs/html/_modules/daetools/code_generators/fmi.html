    def _addUnitDefinition(self, dae_unit):
        dae_bu = dae_unit.baseUnit
        
        unit = fmiUnit()
        bu = fmiBaseUnit()
        unit.name     = str(dae_unit) # *
        unit.baseUnit = bu

        bu.factor = dae_bu.multiplier
        bu.offset = 0.0
        if dae_bu.L != 0:
            bu.m = dae_bu.L
        if dae_bu.M != 0:
            bu.kg = dae_bu.M
        if dae_bu.T != 0:
            bu.s = dae_bu.T
        if dae_bu.C != 0:
            bu.cd = dae_bu.C
        if dae_bu.I != 0:
            bu.A = dae_bu.I
        if dae_bu.O != 0:
            bu.K = dae_bu.O
        if dae_bu.N != 0:
            bu.mol = dae_bu.N
        bu.rad = None
        self.UnitDefinitions.append(unit)

    def _addTypeDefinition(self, var_type):
        t = fmiReal()
        real = fmiSimpleType()
        real.name        = var_type.Name #*
        real.description = None
        real.type        = t #*

        t.quantity           = None
        t.unit               = str(var_type.Unit)
        t.displayUnit        = None
        t.relativeQuantity   = True
        t.min                = var_type.LB
        t.max                = var_type.UB
        t.nominal            = None
        t.unbounded          = True

        self.TypeDefinitions.append(real)
